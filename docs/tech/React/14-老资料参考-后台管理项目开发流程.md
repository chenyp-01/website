- router/wrapperRouteComponent.jsx

  统一处理路由拦截

  ```jsx
  import { Suspense } from 'react'
  import PrivateRoute from './privateRoute'
  export default function wrapperRouteComponent({ title, auth, ...props }) {
      title ? document.title = title : ''
      return (
          <Suspense fallback={<div>loading...</div>}>
              {auth ? <PrivateRoute {...props}></PrivateRoute> : props.element}
          </Suspense>
      )
  }
  
  ```

## 7. API请求的封装

- api/config.js

  ```JS
  import axios from "axios" //第一步：导入axios
  import { message } from 'antd'
  //第二步：创建axios实例
  const service = axios.create({
      baseURL: '/api',
      timeout: 50000//超时时间
  })
  
  // const service2 = axios.create({
  //     baseURL: '/aaa',
  //     timeout: 50000//超时时间
  // })
  
  //第三步 请求拦截
  
  service.interceptors.request.use((config) => {
      // 在发送请求之前做些什么 验证token之类的
      // console.log("请求的数据:", config);
      if (sessionStorage.token) {
          // 不再加common
          config.headers.Authorization = 'Bearer ' + sessionStorage.token
      }
      return config;
  }, (error) => {
      return Promise.reject(error)
  }
  )
  
  // //第四步 响应拦截
  service.interceptors.response.use((response) => {
  
      return response
  },
      (error) => {
  
          const { code } = error;
  
          if (code == "ERR_NETWORK") {
              //@ts-ignore
              message.error("没网");
          } else if (code == 401) {
              //@ts-ignore
              message.error("token过期啦,请重新登陆");
              sessionStorage.clear()
              window.location.reload()
          } else {
              message.error(error?.response?.data?.msg);
          }
          //@ts-ignore
          // return Promise.reject(error);
      }
  )
  export default service;
  ```

通过 hooks 创建的 `contextHolder` 必须插入到子元素节点中才会生效，当你不需要上下文信息时请直接调用。

```jsx
import { message } from 'antd'

const App = () => {

    const handler = () => {
       messageApi.open({

    }
    
    return (
    	<>
        	{ contextHolder }
        </>
    )
}
```

```jsx
export default function BreadcrumbList() {
    const { pathname } = useLocation()  // /notice   /charts/pie
    const [breadCrumbList,setBreadCrumbList] = useState([])

    /**
     *  breadcrumbList = [
     *    {
     *      path: '/dashboard/console',
     *      title: '首页',
     *    },
     *  {
     *      path: '/dashboard',
     *      title: 'dashboard',
     *    },

     *      path: '/dashboard/console',
     *      title: '主控台',
     *    },
     * ]

    //处理数据，获取面包屑导航的数据

    const menuLevel1 = "/" + pathname.split("/")[1]  // /notice  /charts
    const handleBreadCrumbList = () => {
        let list = []
        itemsData.forEach(item => {
            if (item.key === menuLevel1) {
                list.push({
                    path: item.key,
                    title: item.label
                })
                item.children?.forEach(smallItem=>{
                    if (smallItem.key === pathname){
                        list.push({
                            path: smallItem.key,
                            title: smallItem.label
                        })
                    }
                })
            }
        })
        setBreadCrumbList(list)
    }
    useEffect(()=>{
        handleBreadCrumbList()
    },[pathname])

    //渲染面包屑
    const render = ()=>{
        return breadCrumbList.map(item=>(
            <Breadcrumb.Item key={item.path}>
                <NavLink to={item.path}>{item.title}</NavLink>
            </Breadcrumb.Item> 
        )
    )}
    return (
        <Breadcrumb>
            <Breadcrumb.Item>首页</Breadcrumb.Item>

                Dashboard
            </Breadcrumb.Item>

                <NavLink to="/dashboard/console">主控台</NavLink>
```

```js
                // 页签中不显示删除功能
                closable: false,
            }],
            activeKey: '/dashboard/console'
        }
    },
    reducers: {
        addTab: (state, { payload }) => {
            let isRepeat = state.tabsList.some(item => item.key === payload.key)
            if (isRepeat) {
                state.activeKey = payload.key
                return;
            }
            state.tabsList.push(payload)
            state.activeKey = payload.key
            // 需要持久化的数据在改变后更新缓存
            sessionStorage.setItem('tabsStore', JSON.stringify(state))
        },
        toggleTab: (state, { payload }) => {
            state.activeKey = payload
            // 需要持久化的数据在改变后更新缓存
            sessionStorage.setItem('tabsStore', JSON.stringify(state))
        },
        removeTab: (state, { payload }) => {
            // payload:  /notice   /charts/pie
            let index = state.tabsList.findIndex(item => item.key === payload)   // index=5  length=6
            state.tabsList.splice(index, 1)    // length=5
            // 如果是最后一项并且最后一项就是当前项时，当激活项调整为前一个项目
            if (index === state.tabsList.length && payload === state.activeKey) {
                state.activeKey = state.tabsList[index - 1].key
            }
            if (state.activeKey === payload) {
                state.activeKey = state.tabsList[index].key
            }
            // 需要持久化的数据在改变后更新缓存
            sessionStorage.setItem('tabsStore', JSON.stringify(state))
        }
    }
})

export const { addTab, toggleTab, removeTab } = tabsSlice.actions
export default tabsSlice.reducer
```

## 18. Antd中学生管理表格分页的两种方案

第一种： 每次请求后台数据时携带page, 拉取指定页的信息，之后无论页码变化、编辑、删除都需要维护page

第二种： 请求后台不携带page, 一次拉取所有数据，不用维护page,只要提供total和pageSize会自动维护分页

### 8. 删除班级的边界处理

- 删除最后一页唯一的一条数据时，再拉取数据，页码要减1

```js
//判断当前删除的是否为最后一页唯一的一条,
  const isEnd = () => {
    //算法思路： 计算最后一页（当前页）第一条是总条数里的第几个，如果数值和total相等，即为最后一页唯一的一条
    // 13 
    return total === (page - 1) * 4 + 1
  }
  //删除班级
  const onDelete = async (classId) => {
    let res = await _delClass({
      classId
    })
    console.log(res);
    let { code } = res.data
    if (code === 0) {
      message.success("删除成功")
      //判断当前删除的是否为最后一页唯一的一条,并且不是所有数据仅剩的一条
    
      if (isEnd() && total !== 1) {
        setPage(page - 1)
      } else {
        setIsAddAction(!isAddAction)
      }

    } else {
      message.error("删除失败")
    }
  }
```
