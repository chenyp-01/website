### 1. 表单的输入绑定

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <title>Document</title>
  </head>

  <body>
    <div id="app">
      <!-- 单行文本框 -->
      <p>
        单行文本框：<input
          type="text"
          placeholder="请输入信息"
          v-model="msg1"
        />
        结果是：{{msg1}}
      </p>
      <br />
      <!-- 多行文本框 -->
      <p>
        多行文本框： <textarea v-model="msg2"></textarea>
        结果是:{{msg2}}
      </p>
      <br />
      <!-- 单个复选框，绑定到布尔值： -->
      <!-- 复选框不要绑定click,应该绑定@change事件 -->
      <p>
        单个复选框：
        <input type="checkbox" id="checkbox1" v-model="msg3" />
        <label for="checkbox1">结果是：{{msg3}}</label>
      </p>
      <br />

      <!-- 多个复选框，绑定到同一个数组： -->
      <div id="example-3">
        多个复选框：
        <input type="checkbox" id="jack" value="Jack" v-model="list" />
        <label for="jack">Jack</label>
        <input type="checkbox" id="john" value="John" v-model="list" />
        <label for="john">John</label>
        <input type="checkbox" id="mike" value="Mike" v-model="list" />
        <label for="mike">Mike</label>
        ---------------结果是：<span>{{list}}</span>
      </div>
      <br />
      <!-- 单选按钮 -->
      <div id="example-4">
        单选按钮：
        <input type="radio" id="one" value="One" v-model="option1" />
        <label for="one">One</label>
        <input type="radio" id="two" value="Two" v-model="option1" />
        <label for="two">Two</label>
        --------------<span>选中的是: {{option1}}</span>
      </div>
      <br />
      <!-- 选择框（下拉菜单） -->
      <!-- 单选时： -->
      <div id="example-5">
        下拉菜单：
        <select v-model="option2">
          <option disabled value="">请选择</option>
          <option>A</option>
          <option>B</option>
          <option>C</option>
        </select>
        <span>选中结果是: {{option2}}</span>
      </div>
      <br />
      <!-- 多选时 (绑定到一个数组)： -->
      <div id="example-6">
        下拉列表：
        <select multiple style="width: 50px" v-model="option3">
          <option>A</option>
          <option>B</option>
          <option>C</option>
        </select>
        <span>选中结果是:{{option3}}</span>
      </div>
      <br />
      <!-- 选择框（下拉菜单） -->

      <!-- 用 v-for 渲染的动态选项： -->
      动态渲染下拉菜单：
      <select>
        <option v-for="item in userList">{{item}}</option>
      </select>

      <button @click="addUser">添加新用户</button>
    </div>
    <script src="js/vue3.js"></script>
    <script>
      let { createApp } = Vue;
      const app = createApp({
        data() {
          return {
            msg1: "hello", // 单行文本框
            msg2: "多行文本框", // 多行文本框
            msg3: true, // 单个复选框
            list: ["Mike"], // 多个复选框
            // list:[],
            // list:['a','b','c'],
            option1: "", // 单选按钮
            option2: "", // 下拉菜单
            option3: [], // 下拉列表
            userList: ['Green','Mike']
          };
        },
        methods: {
          addUser() {
            this.userList.push('Lucy')
          },
        },
      });
      app.mount("#app");
    </script>
  </body>
</html>

```

### 2. 表单输入绑定的修饰符

表单输入绑定修饰符(v-model修饰符)  

- v-model.lazy    

  在失去输入焦点时，再更新数据

- v-model.number 

  接收的数据会自动转为数值型

- v-model.trim 

  自动去掉数据前后空白字符

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <title>Document</title>
  </head>

  <body>
    <div id="app">
      <!-- 表单输入绑定修饰符(v-model修饰符)  .lazy   .number  .trim -->

      <p>1: <input type="text" v-model="msg1">{{msg1}}</p>
      <!-- lazy: 在失去输入焦点时，再更新数据 -->
      <p>2: <input type="text" v-model.lazy="msg2">{{msg2}}</p>

      <!-- number: 接收的数据会自动转为数值型 -->
      <p>3: <input type="text" v-model="msg3">{{msg3 + 1}}</p>
      <p>4: <input type="text" v-model.number="msg4">{{msg4 + 1}}</p>

      <!-- trim: 自动去掉数据前后空白字符 -->
      <p>5:<input type="text" v-model="msg5" />----{{msg5}}----</p>
      <p>6:<input type="text" v-model.trim="msg6" />----{{msg6}}----</p>
    </div>
    <script src="js/vue3.js"></script>
    <script>
      let { createApp } = Vue;
      const app = createApp({
        data() {
          return {
            msg1: "",
            msg2: "",
            msg3: "",
            msg4: "",
            msg5: "",
            msg6: ""
          };
        },
      });
      app.mount("#app");
    </script>
  </body>
</html>

```
