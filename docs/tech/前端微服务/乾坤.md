1.主应用安装

```
npm install qiankun --save
```

2.主应用改造 main.js

```js
import "./style.css";
import App from "./App.vue";
import { createApp } from "vue";
const app = createApp(App);
import { registerMicroApps, start } from "qiankun";
app.mount("#app");
createApp(App).mount("#app");
const apps = [
  {
    name: "vue-sub",
    entry: "//localhost:7101", // 子应用地址
    container: "#sub-container", // 主应用 HTML 中必须有这个容器
    activeRule: "/vue-sub", // 当前路由为 /vue-sub 时激活子应用
  },
];
registerMicroApps(apps, {
  beforeLoad: (app) => {
    console.log("before load app.name ===>", app.name);
  },
  beforeMount: (app) => {
    console.log("before mount app.name ===>", app.name);
  },
  afterUnmount: (app) => {
    console.log("after unmount app.name ===>", app.name);
  },
});
start(); // 必须调用！不然不会加载
```

5.子应用安装插件

```
npm i vite-plugin-qiankun
```

6.子应用在 `main.js` 中暴露生命周期：

```js
import { createApp } from "vue";
import App from "./App.vue";
import {
  renderWithQiankun,
  qiankunWindow,
} from "vite-plugin-qiankun/dist/helper";
let app;
if (!qiankunWindow.__POWERED_BY_QIANKUN__) {
  createApp(App).mount("#app");
} else {
  renderWithQiankun({
    mount(props) {
      console.log("Vue app mount props:", props);
      app = createApp(App, props);
      app.mount(props.container.querySelector("#app"));
    },
    bootstrap() {
      console.log("Vue app bootstraped");
    },
    update() {
      console.log("Vue app update props:", props);
      // 如果需要更新应用状态，可以在这里处理
    },
    unmount() {
      if (app) {
        app.unmount();
        app = null;
      }
    },
  });
}
```

7.子应用的vite.config.js

```js
// qiankun-sub/vite.config.js
import { defineConfig } from "vite";
import vue from "@vitejs/plugin-vue";
import qiankun from "vite-plugin-qiankun";
export default defineConfig({
  base: "/vue-sub",
  plugins: [
    vue(),
    qiankun("vue-sub", {
      // 子应用名称，与主应用注册的一致
      useDevMode: true,
    }),
  ],
  build: {
    target: "esnext",
    outDir: "dist",
    lib: {
      entry: "src/main.js",
      name: "vue-sub",
      fileName: (format) => `vue-sub.${format}.js`,
      formats: ["umd"],
    },
    rollupOptions: {
      external: ["vue"],
      output: {
        globals: {
          vue: "Vue",
        },
        // 确保qiankun能识别导出的生命周期函数
        format: "umd",
        name: "vue-sub",
        exports: "named",
      },
    },
  },
  server: {
    cors: true,//跨域
    port: 7101,//端口
    headers: {
      "Access-Control-Allow-Origin": "*",
    },
  },
});
```
