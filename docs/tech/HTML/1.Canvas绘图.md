#### 1.认识Canvas

```html
<canvas></canvas>
```

1. canvas标签定义图形，比如图表和其他图像。
2. canvas 标签只是图形容器，您必须使用脚本来绘制图形。
3. 这个 HTML 元素是为了客户端位图图形而设计的。它自己没有行为，但却把 一个绘图 API 展现给客户端 JavaScript 以使脚本能够把想绘制的东西都绘制 到一块画布上。

4. Canvas 绘图 API 定义在通过画布的 getContext() 方法获得的一个“绘图环境”对象上。

#### 2. Canvas基础用法

1. 创建容器

   ```html
   <canvas id="canvas"></canvas>
   ```

2. 添加样式

   ```html
   <canvas id="canvas" style="border:1px solid #aaa; display:block; margin:50px auto;"></canvas>
   ```

3. 设置宽高

   \*\* canvas默认宽高为300px, 150px

   \*\* 注意canvas的宽高不能用css样式设置,因为这样绘制的图形会变形

   ```html
   <canvas
     width="1024"
     height="768"
     id="canvas"
     style="border:1px solid #aaa; display:block; margin:50px auto;"
   ></canvas>
   ```

4. js初始化

   ```js
   window.onload = function () {
     // 获取convas元素
     var canvas = document.getElementById('canvas')

     // 定义绘图的大小
     canvas.width = 1024
     canvas.height = 768

     // 初始化 绘制2d图
     var context = canvas.getContext('2d')
   }
   ```

#### 3. 绘制多种形状

##### 1. 绘制线段

参考： http://blog.csdn.net/clh604/article/details/8536059

- context.moveTo(100,100) 开始位置
- context.lineTo(700,700) 结束位置
- context.stroke() 绘制图
- context.lineWidth 线的宽度
- context.strockStyle 线的颜色

```js
context.strokeStyle = 'rgb(250,0,0)' //描边颜色
context.fillStyle = 'rgb(250,0,0)' //填充颜色
//第一次lineTo的时候和moveTo功能一样
context.moveTo(100, 100)
//之后的lineTo会以上次lineTo的节点为开始
context.lineTo(200, 200)
context.lineTo(200, 100)
//开始新的绘制，不会连接上一个点
context.moveTo(200, 50)
context.lineTo(100, 50)
context.stroke()
```

##### 2. 绘制贝塞尔曲线

quadraticCurveTo(dx,dy,x1,y1)

> 开始点：moveTo(20,20)
> quadraticCurveTo(20,100,200,20) 第一个控制 点，第二个是结束点

提示：二次贝塞尔曲线需要两个点。第一个点是用于二次贝塞尔计算中的控制点，第二个点是曲线的结束点。曲线的开始点是当前路径中最后一个点。如果路径不存在，那么请使用 beginPath() 和 moveTo() 方法来定义开始点。

```js
var canvas = document.getElementById('myCanvas')
var context = canvas.getContext('2d')
context.beginPath()
context.moveTo(20, 20)
context.quadraticCurveTo(20, 100, 200, 20)
context.stroke()
```

##### 3. 画矩形

context.strokeRect(120, 0, 100, 100); // 矩形边框

context.clearRect(50, 50, 240, 120); // 清除矩形区域，橡皮擦功能

```js
context.strokeRect(240, 120, 100, 100)

```

##### 4. 画圆弧

context.arc(x,y,radius,startAngle,endAngle,anticlockwise)

> 1度角的角度 = Math.PI / 180弧度

```
// 圆
context.arc(50, 50, 50, 0, 2*Math.PI , false);
context.fillStyle = 'rgba(255,0,0,0.8)';
context.fill();
context.strokeStyle = 'rgba(255,0,0,1)'
context.stroke();


//1/4圆弧
context.moveTo(100,150);  /*添加，为饼形，不加，为弧形*/
context.arc(100, 150, 50, 0, Math.PI/2 , false);
context.fillStyle = 'rgba(255,0,0,0.8)';
context.fill();
context.strokeStyle = 'rgba(255,0,0,1)'
context.stroke();

```

##### 5. 画图像

context.drawImage( image , 0 ,0 , 50 , 50 )
参数：图像对象，x,y,width,height

```js
var canvas = document.getElementById('canvas')
var context = canvas.getContext('2d')
var image = new Image()

image.src = 'img-lg.jpg'
image.onload = function () {
  context.drawImage(image, 0, 0, 50, 50)
  context.clearRect(0, 0, 150, 150) //擦除某个部分
  context.drawImage(image, 50, 50, 100, 100)
}
```

#### 4. Canvas实现动画的原理

每画一次，都要清除上一次的画布

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <meta http-equiv="X-UA-Compatible" content="ie=edge" />
    <title>Document</title>
    <style>
      canvas {
        background-color: #aaa;
      }
    </style>
  </head>
  <body>
    <canvas width="500" height="400" id="can"></canvas>
    <script>
      let can = document.getElementById('can')
      let cxt = can.getContext('2d')
      let x = 1
      setInterval(() => {
        x++
        cxt.clearRect(0, 0, 500, 400)
        cxt.fillRect(x, 0, 100, 100)
      }, 100)
    </script>
  </body>
</html>
```

#### 5. Canvas中的合成操作

context.globalCompositeOperation 设置合成属性

```js
//这个值与顺序无关，如果与蓝色图形重叠，就将两者的颜色值想加。得到的颜色值的最大取值为255，结果就为白色。
context.globalCompositeOperation = 'lighter'

// 当前图形绘制之处都变成透明
context.globalCompositeOperation = 'destination-out'
```

帮助理解合成操作的案例

```html

  <head>

    <meta http-equiv="X-UA-Compatible" content="IE=edge" />

        // context.globalCompositeOperation = "lighter";

        context.fillRect(50, 50, 100, 50)
        context.closePath()
      }
    </script>
  </body>
```

#### 6. Canvas案例 (理解即可)

##### 1. 图表百分比

```js
<!doctype html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>canvas</title>
    <style>
        canvas {
            background: #e2e2e2;
        }
    </style>
</head>

<body>
    <canvas width="600" height="300" id="chart">
	</canvas>

    <script>
        // colors
        // 圈圈的颜色，百分比颜色，中间空的白的颜色，中间文字的颜色
        var colors = ['#F00', '#900', '#FFF', '#909'];

        // per: 数字,必须, 0-100， 百分比
        // colorIndex: 数字, 必须，数组colors的index值。
        // w:数字,不是必须，默认150，大半径。
        function draw(per, colorIndex, w) {
            var w = w || 150;
            var paper = document.getElementById('chart'); //取得canvas绘图区域
            var chart = paper.getContext('2d'); //2D绘图模式
            var startPoint = Math.PI * 1.75; // 百分比的开始点  Math.PI是一个半圆，Math.PI*2是一个圆
            var endPoint = Math.PI * (2 * (per / 100) + 1.75); // 百分比的结束点   （2+0.95）
            chart.fillStyle = colors[colorIndex]; //设置填充的颜色
            chart.beginPath(); //开始绘图
            chart.moveTo(150, 150); //绘图开始点
            chart.arc(150, 150, w, startPoint, endPoint, false); //绘制圆(弧), (150圆心X坐标,150圆心Y坐标,w圆的半径,startPoint开始点,endPoint结束点,true逆时针);
            chart.fill(); //填充绘制的圆
        }

        function chart(per) {
            draw(100, 1); //画大圆
            draw(per, 0); //画百分比的圆
            draw(100, 2, 100); //画中间的空白圆
            var paper = document.getElementById('chart');
            var chart = paper.getContext('2d');
            chart.font = "4em Arial"; //字体属性
            chart.fillStyle = colors[3] //字体的颜色
            chart.fillText(per + '%', 90, 170); //绘制文字
        }
        chart(60); //执行
    </script>
</body>

</html>
```

##### 2. 刮刮乐

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <title>Document</title>
    <style>
      * {
        margin: 0;
        padding: 0;
      }

      #box {
        width: 300px;
        height: 150px;
        border: 10px dashed #ccc;
        line-height: 130px;
        text-align: center;
        font-size: 40px;
        box-sizing: border-box;
        margin: 50px;
      }

      #cvs {
        position: absolute;
        top: 50px;
        left: 50px;
        /* position:relative;
	top:-200px;
	left:50px;
	*/
      }
    </style>
  </head>

  <body>
    <div id="box">谢谢惠顾</div>
    <!-- cannvas是h5新增标签，是一个用来展示绘图效果的标签，和img类似，都是行内快元素 -->
    <canvas id="cvs" height="150" width="300"></canvas>

    <script>
      var cvs = document.getElementById('cvs')
      var box = document.getElementById('box')

      var arr = [
        '谢谢惠顾',
        '谢谢惠顾',
        '谢谢惠顾',
        '谢谢惠顾',
        '一等奖',
        '二等奖',
        '二等奖',
        '三等奖',
        '三等奖',
        '三等奖',
      ]
      var i = Math.floor(Math.random() * 10) //0-9
      box.innerHTML = arr[i]

      // 每个canvas元素都有一个对应的context对象（上下文对象）
      // Context API定义在context对象上面，所需要获取这个对象；
      // 方法使用getContext方法

      // 获取context   getContext是canvas对象的方法
      // getContext方法指定参数2d，表示生成2D图案。如果参数是webg1，就表示生成3D图像；
      var c = cvs.getContext('2d')
      // 填充色
      c.fillStyle = '#ccc'
      // 规定形状 位置 和尺寸填充方式
      c.fillRect(0, 0, cvs.width, cvs.height)

      function draw(e) {
        // canvas中的合成操作
        c.globalCompositeOperation = 'destination-out'
        c.beginPath() //开始路径绘制  为了防止重绘之前路径 ，所以把之前的路径清除掉
        c.arc(e.clientX - cvs.offsetLeft, e.clientY - cvs.offsetTop, 20, 0, 2 * Math.PI, false)
        c.fill()
      }
      // 事件监听
      cvs.addEventListener(
        'mousedown',
        function () {
          cvs.addEventListener('mousemove', draw, false)
        },
        false,
      )
      document.addEventListener(
        'mouseup',
        function () {
          cvs.removeEventListener('mousemove', draw, false)
        },
        false,
      )
    </script>
  </body>
</html>
```

##### 3. 图像缩放

```html
<!DOCTYPE html>
<html>
  <head lang="en">
    <meta charset="UTF-8" />
    <title></title>
  </head>
  <body style="background: black;">
    <canvas id="canvas" style="display:block;margin:0 auto;border:1px solid #aaa;">
      您的浏览器尚不支持canvas
    </canvas>
    <input
      type="range"
      id="scale-range"
      min="0.5"
      max="3.0"
      step="0.01"
      value="1.0"
      style="display:block;margin:20px auto;width:800px"
    />

    <script>
      var canvas = document.getElementById('canvas')
      var context = canvas.getContext('2d')
      var slider = document.getElementById('scale-range')
      var image = new Image()

      window.onload = function () {
        canvas.width = 1152
        canvas.height = 768

        var scale = slider.value
        image.src = 'img-lg.jpg'
        image.onload = function () {
          drawImageByScale(scale)
          slider.onmousemove = function () {
            scale = slider.value
            drawImageByScale(scale)
          }
        }
      }

      function drawImageByScale(scale) {
        var imageWidth = 1152 * scale
        var imageHeight = 768 * scale

        x = canvas.width / 2 - imageWidth / 2
        y = canvas.height / 2 - imageHeight / 2

        context.clearRect(0, 0, canvas.width, canvas.height)
        context.drawImage(image, x, y, imageWidth, imageHeight)
      }
    </script>
  </body>
</html>
```
